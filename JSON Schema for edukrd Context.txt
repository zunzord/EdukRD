## JSON Schema for Android App Context

This JSON Schema defines the structure for capturing your Android project's architecture, navigation, models, repositories, screens, viewModels, and file structure.

```jsonschema
{
  "$schema": "http://json-schema.org/draft-07/schema#",
  "title": "AndroidAppContext",
  "type": "object",
  "properties": {
    "appName": {"type": "string"},
    "architecture": {
      "type": "object",
      "properties": {
        "pattern": {"type": "string"},
        "modules": {"type": "array", "items": {"type": "string"}}
      },
      "required": ["pattern", "modules"]
    },
    "navigation": {
      "type": "array",
      "items": {"type": "object", "properties": {"route": {"type": "string"}, "args": {"type": "array", "items": {"type": "string"}}}, "required": ["route"]}
    },
    "models": {"type": "object", "additionalProperties": {"type": "object", "additionalProperties": {"type": "string"}}},
    "repositories": {"type": "array", "items": {"type": "string"}},
    "screens": {"type": "array", "items": {"type": "string"}},
    "viewModels": {"type": "array", "items": {"type": "string"}},
    "fileStructure": {"type": "array", "items": {"type": "string"}}
  },
  "required": ["appName", "architecture", "navigation", "models", "repositories", "screens", "viewModels", "fileStructure"]
}
```
